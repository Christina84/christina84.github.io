/*basic select query syntax*/
SELECT column1, column2, ...
FROM table_name;
WHERE conditionsl

/*select distinct returns no duplicate table values*/
SELECT DISTINCT column1, column2, ...
FROM table_name;

/*insert values into table syntax*/
INSERT INTO table_name (column1, column2, column3, ...)
VALUES (value1, value2, value3, ...);

/*update table values*/
UPDATE table_name
SET column1 = value1, column2 = value2, ...
WHERE condition;

/*delete rows from a table*/
DELETE FROM table_name WHERE condition;

Personal Example

/*gets all transactions from for store between one date and another*/
create function dbo.GetTransactionCertainDate (@BeginDate as datetime, @EndDate as datetime)
returns table
as
begin
select CustomerID, DateOfTransaction
from Transactions
where DateOfTransaction between @BeginDate and @EndDate
end
go
