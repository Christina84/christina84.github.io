Cheat Sheet!

<br> /*adds a line break*/
<!doctype html> /*specifies document as html file*/
<p> /*paragraph element*/
<h1-h6> /*headers h1 being the largest header*/
<head> /*title/things that arent displayed */
<div> /*style block content with css*/
<span> /*style inline content with css*/
<ul> /*unordered list*/
<ol> /*ordered list*/
<li> /*list item*/
<b> /*bold text*/
<i> /*italicized*/
<cite> /*citation*/
<font size= 1> /*sets font size, usually done in externam CSS*/
<a href="URL">clickable text> /*link*/
<img src="URL"> /*image from separate file*/
<form> /*creates a form*/
<table> /*creates a table*/
<tr> /*row*/
<td> /*cell*/
<th> /*table header*/
